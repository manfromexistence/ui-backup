## Usage

```tsx
import { Tspopover, TspopoverTrigger, TspopoverContent } from "@/components/ui/tspopover";

<Tspopover>
  <TspopoverTrigger>Trigger Button</TspopoverTrigger>
  <TspopoverContent>
    {/* Content for the initial popover view */}
    <ul>
      <li onClick={handleListItemClick}>Item 1</li>
      <li onClick={handleListItemClick}>Item 2</li>
    </ul>
  </TspopoverContent>
</Tspopover>
